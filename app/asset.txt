// Define the props for the Card component
type CardProps = {
    numberTitle: string;
    unit: string;
    text: string;
    backgroundImage: string;
    index: number;
    cardie: number;
};

function CardStack({ numberTitle, text, unit, backgroundImage, index, cardie }: CardProps) {
    const containerRef = useRef<HTMLDivElement>(null);

    useEffect(() => {
        const container = containerRef.current;
        const items = container?.querySelectorAll(".stack-item");

        const handleScroll = () => {
            const scrollTop = window.scrollY;

            items?.forEach((item, index) => {
                 const offsetFactor = 200; //Adjust the stacking effect 
                const topPosition = index === 1 ? 0 : (index - 1) * (cardie / 1.5);
                const offset = scrollTop - index * 500;

                if (offset >= 0) {
                    (item as HTMLElement).style.transform = `translate(-50%, ${-offset}px)`; // 
                } else {
                    (item as HTMLElement).style.transform = `translate(-50%, ${index * 10
                        }px)`;
                }
            });
        };

        window.addEventListener("scroll", handleScroll);
        return () => window.removeEventListener("scroll", handleScroll);
    }, []);




    const topPosition = index === 1 ? 0 : (index - 1) * (cardie / 1.5);

    return (
        <div ref={containerRef} className="stack-container h-[500px]"
        >
            <div
                className="absolute sm:w-1/2 md:w-10/12 lg:w-10/12 mx-auto rounded-xl my-10 stack-item"
                style={{
                    backgroundImage: `url(${backgroundImage})`,
                    backgroundSize: "cover cover",
                    backgroundPosition: "center center",
                    backgroundRepeat: 'no-repeat',
                    height: '500px',
                    borderRadius: "50px",
                    left: "50%",
                    transform: "translateX(-50%)",
                    transition: "transform 0.3s ease",
                    // top: `${topPosition}px`,
                    top:"2000px"
                    
                }}
            >
            <div className="sm:my-20 lg:my-0">
                <div className="flex justify-start items-center">
                    <div className="pl-14 py-5 lg:w-8/12 text-white ">
                        <h1 className={`lg:text-[80px] sm:text-[30px] md:text-[60px] font-[rubik] font-extrabold  ${index === 2 ? 'text-white' : 'text-[#063A4F]'
                            } ${index === 6 ? 'pt-12' : 'pt-0'}`}>
                            {numberTitle}
                        </h1>
                        <p className="lg:text-[60px] sm:text-[25px] md:text-[40px] font-[rubik] font-extrabold pb-5">
                            {unit}
                        </p>
                        <p className="lg:text-2xl sm:text-base font-[inter] font-normal pb-12">
                            {text}
                        </p>
                    </div>
                </div>
            </div>
        </div>
        </div >

    );
}

export default CardStack;
